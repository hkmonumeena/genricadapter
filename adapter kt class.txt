    RecyclerView.Adapter< ${NAME}.UserViewHolder>() {

        LayoutInflater.from(parent.context).inflate(R.layout.LAYOUTNAMEHERE, parent, false),
        parent.context
    )

    override fun getItemCount() = modelDataList.size

    override fun onBindViewHolder(holder: UserViewHolder, position: Int) {
        val modelObj: ModelNameHere = modelDataList[position]

        holder.bind(modelDataList[position], position)
    }

    class UserViewHolder(view: View, context: Context) : RecyclerView.ViewHolder(view) {
        val context = context
        private val productTitle = view.list_description

        fun bind(modelObj: ModelNameHere, pos: Int) {
            productTitle.text = modelObj.Status
        }
